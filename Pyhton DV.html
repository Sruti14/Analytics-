from pywaffle import Waffle
category_expenditure = df.groupby("Payment")["Total"].sum()

total_tiles = 50
category_proportions =(category_expenditure/ category_expenditure.sum()* total_tiles).round().astype(int)

fig= plt.figure(
    FigureClass= Waffle,
    rows= 15,
    values = category_proportions,
    colors=["red", "blue", "green"],
    title = {'label': 'Payment method vs Total amount', 'loc': 'center', 'fontsize': 16},
    labels= [f"{label} ($({value})" for label, value in category_expenditure.items()],
    legend= {'loc': 'upper left', 'bbox_to_anchor':(1.5,1)}
)

plt.show()
